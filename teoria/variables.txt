-------------------------------------------------------------------------------------------------------
- Declarar una variable: ¡Sin espacios entre el nombre, el = y el valor!
-------------------------------------------------------------------------------------------------------

nombre="Juanan"
num=10

-------------------------------------------------------------------------------------------------------
- Usar una variable:
-------------------------------------------------------------------------------------------------------

echo "Hola, $nombre"

-------------------------------------------------------------------------------------------------------
- Variables locales: Solo existen dentro del script o función donde se declaran.
-------------------------------------------------------------------------------------------------------

contador=5

-------------------------------------------------------------------------------------------------------
- Variables de entorno: Están disponibles para subprocesos y otros comandos.
-------------------------------------------------------------------------------------------------------

export PATH=$PATH:/nuevo/directorio

-------------------------------------------------------------------------------------------------------
- Operaciones aritméticas: 
-------------------------------------------------------------------------------------------------------

Usa $(( ... )) para cálculos.

a=5
b=3
suma=$((a + b))
echo "Suma: $suma"

-------------------------------------------------------------------------------------------------------
- Variables especiales (predefinidas por bash)
-------------------------------------------------------------------------------------------------------

Variable	Significado

$0		Nombre del script
$1, $2...	Argumentos del script
$#		Número de argumentos
$@		Todos los argumentos (como array)
$?		Código de salida del último comando
$$		PID del proceso actual
$!		PID del último proceso en segundo plano
